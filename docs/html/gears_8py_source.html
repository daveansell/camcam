64  # generates an involute curve from a circle of radius r up to theta_max radians
65  # with a specified number of steps
66  def gears_generate_involute( self, r, r_max, theta_max, steps=30 ):
67  dtheta = theta_max / float(steps)
68  x = []
69  y = []
70  theta = []
71  rlast = r;
72  for i in range( 0, steps+1 ):
73  c = math.cos( i*dtheta )
74  s = math.sin( i*dtheta )
75  tx = r*( c + i*dtheta*s )
76  ty = r*( s - i*dtheta*c )
77  d = math.sqrt(tx*tx+ty*ty)
78  if d > r_max:
79  a = (r_max-rlast)/(d-rlast)
80  tx = x[-1]*(1.0-a) + tx*a
81  ty = y[-1]*(1.0-a) + ty*a
82  ttheta = theta[-1]*(1.0-a) + math.atan2( ty, tx )*a
83  x.append( tx )
84  y.append( ty )
85  theta.append( ttheta )
