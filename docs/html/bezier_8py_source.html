1489  # Look up what the binomial coefficient is for pair {n,k}
1490  private float binomials(int n, int k) {
1491  while(n >= binomial_coefficients.length) {
1492  int s = binomial_coefficients.length;
1493  float[][] update_coefficients = new float[s+1][];
1494  arrayCopy(binomial_coefficients,0,update_coefficients,0,s);

1505  binomial_coefficients = update_coefficients;
1506  }
