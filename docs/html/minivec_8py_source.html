976  - (sa * sj - si * sb) * (sg * sp - so * sh)

978  + (se * sj - si * sf) * (sc * sp - so * sd)
979  - (se * sn - sm * sf) * (sc * sl - sk * sd)
980  + (si * sn - sm * sj) * (sc * sh - sg * sd))

1101  if fabs(cosHalfTheta) > 1.0 - epsilon:
1102  return q1
